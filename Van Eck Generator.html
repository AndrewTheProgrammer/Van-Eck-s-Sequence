def vaneckine(startnum,numbers=10000):
    backwardslist=[startnum]
    print(str(startnum))
    i=1
    while (i<numbers):
        try:
            backwardslist.insert(0,backwardslist[1:].index(backwardslist[0])+1)
        except ValueError:
            backwardslist.insert(0,0)
        print(str(backwardslist[0]))
        i+=1
